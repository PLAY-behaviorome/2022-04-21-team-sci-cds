Collection site demographics

Rick Gilmore

2022-04-13 15:21:07

Purpose

This document generates demographic figures for the PLAY data sites. It draws upon code and data in the PLAY-behaviorome/site-demographics repository.

The source data comes from a directory hidden to git for the time being.

Set-up

Race

race <-
  readr::read_csv("csv/collection-by-county-race.csv")
## Rows: 36 Columns: 11
##  Column specification 
## Delimiter: ","
## chr (3): NAME, Institution, SiteCode
## dbl (8): am_indian, asian, black, native_hawaiian, one_race, other, total, w...
## 
##  Use `spec()` to retrieve the full column specification for this data.
##  Specify the column types or set `show_col_types = FALSE` to quiet this message.
race_df <- race %>%
  dplyr::rename(., county = NAME) %>%
  dplyr::mutate(., multiple = total - one_race) %>%
  dplyr::rename(., site_code = SiteCode) %>%
  dplyr::mutate(
    .,
    am_indian = am_indian / total,
    asian = asian / total,
    black = black / total,
    native_hawaiian = native_hawaiian / total,
    other = other / total,
    white = white / total,
    multiple = multiple / total
  ) %>%
  dplyr::select(
    .,
    site_code,
    county,
    am_indian,
    asian,
    black,
    native_hawaiian,
    other,
    white,
    multiple
  ) %>%
  dplyr::arrange(., white) %>%
  dplyr::mutate(., white_ranking = 1:length(white)) %>%
  tidyr::pivot_longer(
    .,
    cols = c(
      'am_indian',
      'asian',
      'black',
      'native_hawaiian',
      'other',
      'white',
      'multiple'
    ),
    names_to = "race",
    values_to = "p_pop"
  ) %>%
  dplyr::arrange(., white_ranking) 
race_df %>%
  ggplot(.) +
  aes(fill = race,
      x = white_ranking,
      y = p_pop) +
  geom_bar(position = "fill", stat = "identity") +
  theme_classic() +
  theme(axis.text.x = element_blank()) +
  theme(axis.ticks.x = element_blank()) +
  theme(legend.title = element_blank()) +
  xlab("") +
  ylab("p(pop)")

Spanish-speakers

spanish <- readr::read_csv("csv/collection-by-county-spanish.csv")
## Rows: 216 Columns: 3
##  Column specification 
## Delimiter: ","
## chr (2): NAME, pop
## dbl (1): n
## 
##  Use `spec()` to retrieve the full column specification for this data.
##  Specify the column types or set `show_col_types = FALSE` to quiet this message.
spanish_df <- spanish %>%
  dplyr::rename(.,
                county = NAME,
                lang_spoken = pop,
                p_speakers = n) %>%
  tidyr::pivot_wider(.,
                     id_cols = c('county', 'p_speakers'),
                     names_from = lang_spoken,
                     values_from = p_speakers,
                     values_fn = mean) %>%
  dplyr::select(., -c('English_only', 'Spanish', 'Other')) %>%
  dplyr::rename(., English_only = p_English_only,
                Spanish = p_Spanish,
                Other = p_Other) %>%
  dplyr::arrange(., Spanish) %>%
  dplyr::mutate(., span_ranking = 1:length(Spanish)) %>%
  tidyr::pivot_longer(., cols = -c('county', 'span_ranking'), names_to = "language", values_to = "p_pop")
spanish_df %>%
  ggplot(.) +
  aes(fill = language,
      x = span_ranking,
      y = p_pop) +
  # geom_bar(position = "fill", stat = "identity") +
  geom_col() +
  theme_classic() +
  theme(axis.text.x = element_blank()) +
  theme(axis.ticks.x = element_blank()) +
  theme(legend.title = element_blank()) +
  xlab("") +
  ylab("p(pop)")

Education

ed <- readr::read_csv("csv/collection-by-county-education.csv")
## Rows: 36 Columns: 7
##  Column specification 
## Delimiter: ","
## chr (1): NAME
## dbl (6): total, lt_hs, hs_grad, some_coll, ba, ba_plus
## 
##  Use `spec()` to retrieve the full column specification for this data.
##  Specify the column types or set `show_col_types = FALSE` to quiet this message.
ed_df <- ed %>%
  dplyr::rename(., county = NAME) %>%
  dplyr::mutate(
    .,
    lt_hs = lt_hs / total,
    hs_grad = hs_grad / total,
    some_coll = some_coll / total,
    ba = ba / total,
    ba_plus = ba_plus / total
  ) %>%
  dplyr::arrange(., lt_hs) %>%
  dplyr::mutate(., lt_hs_index = 1:length(lt_hs)) %>%
  tidyr::pivot_longer(
    .,
    cols = c('lt_hs', 'hs_grad', 'some_coll', 'ba', 'ba_plus'),
    names_to = "education",
    values_to = "p_pop"
  ) %>%
  dplyr::select(.,-c('total')) %>%
  dplyr::mutate(., education = factor(
    education,
    levels = c('lt_hs', 'hs_grad', 'some_coll', 'ba', 'ba_plus')
  ))
ed_df %>%
  ggplot(.) +
  aes(fill = education,
      x = lt_hs_index,
      y = p_pop) +
  geom_col() +
  theme_classic() +
  theme(axis.text.x = element_blank()) +
  theme(axis.ticks.x = element_blank()) +
  theme(legend.title = element_blank()) +
  xlab("") +
  ylab("p(pop)")

## Delimiter: ","
## chr (1): NAME
## dbl (5): total, lt_25K, b25_50K, b50_100K, gt_100K
## 
##  Use `spec()` to retrieve the full column specification for this data.
##  Specify the column types or set `show_col_types = FALSE` to quiet this message.
income_df <- income %>%
  dplyr::rename(., county = NAME) %>%
  dplyr::mutate(
    .,
    lt_25K = lt_25K / total,
    b25_50K = b25_50K / total,
    b50_100K = b50_100K / total,
    gt_100K = gt_100K / total
  ) %>%
  dplyr::arrange(., lt_25K) %>%
  dplyr::mutate(., lt_25K_index = 1:length(lt_25K)) %>%

  dplyr::select(., -c('total')) %>%
  dplyr::mutate(., education = ordered(income, levels = c(
    'lt_25K', 'b25_50K', 'b50_100K', 'gt_100K'
  )))
income_df %>%
  ggplot(.) +
  aes(fill = income,
      x = lt_25K_index,
